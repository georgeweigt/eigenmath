\section*{Introduction}

In the following examples, {\tt\color{blue}blue text} is user input.

\bigskip
{\bf Example 1.} Use a caret for exponent.
{\color{blue}
\begin{verbatim}
212^17
\end{verbatim}}
$3529471145760275132301897342055866171392$

\bigskip
{\bf Example 2.} Parentheses are required for negative exponents.
{\color{blue}
\begin{verbatim}
x^(-2)
\end{verbatim}}
$\displaystyle\frac{1}{x^2}$

\bigskip
{\bf Example 3.} Find the derivative of $\sin(x)$.
{\color{blue}
\begin{verbatim}
d(sin(x),x)
\end{verbatim}}
$\cos(x)$

\bigskip
{\bf Example 4.} Show that $(XP-PX)\psi=i\hbar\psi$.
{\color{blue}
\begin{verbatim}
X(f) = x f
P(f) = -i hbar d(f,x)
X(P(psi(x))) - P(X(psi(x)))
\end{verbatim}}
$i\hbar\psi(x)$

\bigskip
{\bf Example 5.} Find $(XP-PX)\psi$ in three dimensions.
{\color{blue}
\begin{verbatim}
X(f) = outer((x,y,z),f)
P(f) = -i hbar d(f,(x,y,z))
X(P(psi(x,y,z))) - P(X(psi(x,y,z)))
\end{verbatim}}
$\begin{bmatrix}
i\hbar\psi(x,y,z) & 0 & 0
\\[1ex]
0 & i\hbar\psi(x,y,z) & 0
\\[1ex]
0 & 0 & i\hbar\psi(x,y,z)
\end{bmatrix}
$
