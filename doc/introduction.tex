\section*{Introduction}

Arithmetic operators have the expected precedence of
multiplication and division before addition and subtraction.
Subexpressions in parentheses have highest precedence.

\begin{center}
\begin{tabular}{clll}
{\it Math} & & {\it Eigenmath} & {\it Comment}
\\
\\
$a=b$ & & \verb$a == b$ & {\it test for equality}
\\[1ex]
$-a$ & & {\tt -a} & {\it negation}
\\[1ex]
$a+b$ & & {\tt a+b} & {\it addition}
\\[1ex]
$a-b$ & & {\tt a-b} & {\it subtraction}
\\[1ex]
$ab$ & & {\tt a b} & {\it multiplication, also} \verb$a*b$
\\
\\

In the following examples, {\tt\color{blue}blue text} is user input.

\bigskip
{\bf Example 1.} Use a caret for exponent.
{\color{blue}
\begin{verbatim}
212^17
\end{verbatim}}
$3529471145760275132301897342055866171392$

\bigskip
{\bf Example 2.} Parentheses are required for negative exponents.
{\color{blue}
\begin{verbatim}
212^(-17)
\end{verbatim}}
$\frac{1}{3529471145760275132301897342055866171392}$

\bigskip
{\bf Example 3.} Symbols are defined with an equals sign.
{\color{blue}
\begin{verbatim}
N = 212^17
N
\end{verbatim}}
$N=3529471145760275132301897342055866171392$

\bigskip
{\bf Example 4.} Solve for $x$ in $Ax=b$ where
$A=\begin{pmatrix}3 & 7\\1 & -9\end{pmatrix}$ and
$b=\begin{pmatrix}16\\-22\end{pmatrix}$.
{\color{blue}
\begin{verbatim}
A = ((3,7),(1,-9))
b = (16,-22)
dot(inv(A),b)
\end{verbatim}}
$\begin{pmatrix}-\frac{5}{17}\\[1ex]\frac{41}{17}\end{pmatrix}$

\bigskip
{\bf Example 5.} Show that $(XP-PX)\psi=i\hbar\psi$.
{\color{blue}
\begin{verbatim}
X(f) = x f
P(f) = -i hbar d(f,x)
X(P(psi(x))) - P(X(psi(x)))
\end{verbatim}}
$i\hbar\psi(x)$

\bigskip
{\bf Example 6.} Count the number of poker hands.
{\color{blue}
\begin{verbatim}
choose(52,5)
\end{verbatim}}
$2598960$
