\section*{Introduction}

Eigenmath was created for doing physics problems,
so here is an example from quantum mechanics.

\bigskip
Let $X$ and $P$ be the operators
\begin{equation*}
X=x\qquad P=-i\hbar\frac{\partial}{\partial x}
\end{equation*}

Show that
\begin{equation*}
(XP-PX)\psi(x,t)=i\hbar\psi(x,t)
\end{equation*}

Eigenmath code
{\color{blue}
\begin{verbatim}
X(f) = x f
P(f) = -i hbar d(f,x)
X(P(psi(x,t))) - P(X(psi(x,t)))
\end{verbatim}}

Result

\bigskip
$i\hbar\psi(x,t)$

\bigskip
In three dimensions (symbol $\otimes$ is outer product, $\nabla$ is gradient)
\begin{equation*}
X=\begin{pmatrix}x\\y\\z\end{pmatrix}\otimes\qquad P=-i\hbar\nabla
\end{equation*}

Eigenmath code
{\color{blue}
\begin{verbatim}
X(f) = outer((x,y,z),f)
P(f) = -i hbar d(f,(x,y,z))
X(P(psi(x,y,z,t))) - P(X(psi(x,y,z,t)))
\end{verbatim}}

Result

\begin{equation*}
\begin{pmatrix}
i\hbar\psi(x,y,z,t)&0&0\\[1ex]
0&i\hbar\psi(x,y,z,t)&0\\[1ex]
0&0&i\hbar\psi(x,y,z,t)
\end{pmatrix}
\end{equation*}

In momentum space
\begin{equation*}
X=i\hbar\frac{\partial}{\partial p}
\qquad
P=p
\end{equation*}

Eigenmath code
{\color{blue}
\begin{verbatim}
X(f) = i hbar d(f,p)
P(f) = p f
X(P(psi(p,t))) - P(X(psi(p,t)))
\end{verbatim}}

Result

\bigskip
$i\hbar\psi(p,t)$

\bigskip
In three dimensions
\begin{equation*}
X=i\hbar\nabla_p
\qquad
P=\begin{pmatrix}p_x\\p_y\\p_z\end{pmatrix}\otimes
\end{equation*}

Eigenmath code
{\color{blue}
\begin{verbatim}
X(f) = i hbar d(f,(px,py,pz))
P(f) = outer((px,py,pz),f)
X(P(psi(px,py,pz,t))) - P(X(psi(px,py,pz,t)))
\end{verbatim}}

Result

\begin{equation*}
\begin{pmatrix}
i\hbar\psi(p_x,p_y,p_z,t)&0&0\\[1ex]
0&i\hbar\psi(p_x,p_y,p_z,t)&0\\[1ex]
0&0&i\hbar\psi(p_x,p_y,p_z,t)
\end{pmatrix}
\end{equation*}
