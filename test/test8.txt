pie = float(pi)

yhat = zero(N)

for(k,1,N,
  M = (1 + x[k])/2 + 2/(1 + x[k]),
  yhat[k] = pie/s alpha^2 M (hbar c)^2 10^40 "picobarn" / meter^2
)

"Predicted cross section"

yhat

"Coefficient of determination (R-squared)"

ybar = 1/N sum(k,1,N,y[k])
SSE = sum(k,1,N,(y[k] - yhat[k])^2)
SST = sum(k,1,N,(y[k] - ybar)^2)
1 - SSE/SST
